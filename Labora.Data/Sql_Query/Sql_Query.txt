CREATE TABLE Country 
(Id int identity(1,1) Primary key ,
Name varchar(255),
Active int default 1,
unique(Name)
)

go

CREATE PROCEDURE [dbo].[SP_ListCountry]
AS
BEGIN
	SELECT Id, Name FROM Country WHERE Active = 1 order by Name Asc
END

go

CREATE TABLE Users 
(Id int identity(1,1) Primary key ,
Email varchar(255),
Password varchar(255),
DateCreate datetime,
Active int default 1,
unique(Email)
)

go
